Phase 10.5 — (旧) 書類自動判定 AND 化計画 → Phase 11 対象グループに統合済 廃止

背景: 現状（lib/document-applicability.ts 43–65行）は「required タグの source_field のうち 1 つでも 該当（isFieldApplicable が true）なら書類は対象」という OR 判定。 last_name / first_name はゲート未定義 → 常に true → これらが required タグに 1 つでもあれば 全社員が対象 になってしまう。

採用方針（ユーザー合意済）: 「ゲート付きの必須タグが 1 つでもあれば、その全てに該当する社員のみ対象」とする AND 判定。氏名等のゲート無しタグは判定から除外（無視）する。

新ロジック（擬似コード）

function isDocumentApplicable(template, employee, customFieldGates) {
  const requiredEmployeeTags = template.mapping
    .filter(m => m.required && m.source_type === 'employee');

  // ゲート定義のあるタグだけ抽出（CORE_FIELD_GATES または customFieldGates にエントリあり）
  const gatedRequired = requiredEmployeeTags.filter(t => hasGateDefinition(t.source_field, customFieldGates));

  if (gatedRequired.length === 0) {
    // ゲート付き必須タグが 0 個 = 全員共通書類（従来どおり全員対象）
    return true;
  }

  // AND: ゲート付き必須タグの「全て」に該当する社員のみ対象
  return gatedRequired.every(t => isFieldApplicable(employee, t.source_field, customFieldGates));
}

function hasGateDefinition(fieldName, customFieldGates) {
  return (CORE_FIELD_GATES[fieldName]?.length ?? 0) > 0
      || (customFieldGates?.get(fieldName)?.length ?? 0) > 0;
}
